11 // To be called WITHIN a given parallel loop. First argument is the parameters to 'capture',
12 // corresponding to the 'firstprivate' in OPENMP.
13 // Usually either '=' to capture all necessary variables, or a list of arguments
14 // declared within the 'with_arguments' macro for a more precise list
15 // The second argument is the actual code to execute. That code should be formulated like
16 // the body of a normal function, e.g. return _SUCCESS_ at the end.
17 // 'private' arguments in OPENMP can be emulated by just making them local to the function.
18 // Special care: A list of private arguments needs to be surrounded by the special macro
19 // 'declare_list_of_variables-inside_parallel_region' due to the peculiarities
20 // of the C++ preprocessor macros. See examples e.g.
21 // in source/perturbations.c, source/lensing.c, or tools/hypershperical.c
22 #define class_run_parallel(arg1, arg2) future_output.push_back(task_system.AsyncTask([arg1] () {arg2}));
23 // The mutable version allows one to change variables outside the scope of the parallel region
24 // Be careful, this is very dangerous, and you should be sure that you don't access any
25 // variables you edit during the parallel region outside of it, as that value will be random
26 #define class_run_parallel_mutable(arg1, arg2) future_output.push_back(task_system.AsyncTask([arg1] () mutable {arg2}));

67 class NotificationQueue {
68 public:
69  bool TryPop(std::function<void()>& x) {
70  std::unique_lock<std::mutex> lock(mutex_, std::try_to_lock);
71  if (!lock || queue_.empty()) {
72  return false;
73  }
74  x = std::move(queue_.front());
75  queue_.pop_front();
76  return true;
77  }
78 
79  bool Pop(std::function<void()>& x) {
80  std::unique_lock<std::mutex> lock(mutex_);
81  while (queue_.empty() && !done_) {
82  ready_.wait(lock);
83  }
84  if (queue_.empty()) {
85  return false;
86  }
87  x = std::move(queue_.front());
88  queue_.pop_front();
89  return true;
90  }
91 
92  template<typename F>
93  bool TryPush(F&& f) {
94  {
95  std::unique_lock<std::mutex> lock(mutex_, std::try_to_lock);
96  if (!lock) {
97  return false;
98  }
99  queue_.emplace_back(std::forward<F>(f));
100  }
101  ready_.notify_one();
102  return true;
103  }
104 
105  template<typename F>
106  void Push(F&& f) {
107  {
108  std::unique_lock<std::mutex> lock(mutex_);
109  queue_.emplace_back(std::forward<F>(f));
110  }
111  ready_.notify_one();
112  }
113 
114  void Done() {
115  {
116  std::unique_lock<std::mutex> lock(mutex_);
117  done_ = true;
118  }
119  ready_.notify_all();
120  }
121 private:
122  std::deque<std::function<void()>> queue_;
123  bool done_ = false;
124  std::mutex mutex_;
125  std::condition_variable ready_;
126 };

128 class TaskSystem {
129 public:
130  TaskSystem(unsigned int count = GetNumThreads())
131  : count_(count)
132  , index_(0)
133  , queues_{count_} {
134  for (unsigned int n = 0; n < count_; ++n) {
135  threads_.emplace_back([&, n]{ Run(n); });
136  }
137  }
138 
139  ~TaskSystem() {
140  for (auto& e : queues_) e.Done();
141  for (auto& e : threads_) e.join();
142  }

144  static unsigned int GetNumThreads() {
145  unsigned int number_of_threads = std::thread::hardware_concurrency();
146  for (const std::string& env_var_name : {"OMP_NUM_THREADS", "SLURM_CPUS_PER_TASK"}) {
147  if (char* s = std::getenv(env_var_name.c_str())) {
148  int threads = std::atoi(s);
149  if ((threads > 0) && (threads <= 8192)) {
150  number_of_threads = threads;
151  break;
152  }
153  }
154  }
155  return number_of_threads;
156  }

158  template <typename F>
159  void Async(F&& f) {
160  auto i = index_++;
161  for (unsigned int n = 0; n < count_; ++n) {
162  if (queues_[(i + n) % count_].TryPush(std::forward<F>(f))) {
163  return;
164  }
165  }
166  queues_[i % count_].Push(std::forward<F>(f));
167  }

169  template<typename F>
170  std::future<typename std::result_of<F()>::type> AsyncTask(F&& f) {
171  using return_type = typename std::result_of<F()>::type;
172  auto task = std::make_shared<std::packaged_task<return_type()>>(f);
173  std::future<return_type> res = task->get_future();
174 
175  auto work = [task](){ (*task)(); };
176  unsigned int i = index_++;
177  for(unsigned int n = 0; n < count_; ++n) {
178  if(queues_[(i + n) % count_].TryPush(work)){
179  return res;
180  }
181  }
182  queues_[i % count_].Push(work);
183 
184  return res;
185  }
186 
187  template<typename F, typename... Args>
188  std::future<typename std::result_of<F(Args...)>::type> AsyncTask(F&& f, Args&&... args) {
189  using return_type = typename std::result_of<F(Args...)>::type;
190  auto task = std::make_shared<std::packaged_task<return_type()>>(std::bind(std::forward<F>(f), std::forward<Args>(args)...));
191  std::future<return_type> res = task->get_future();
192 
193  auto work = [task](){ (*task)(); };
194  unsigned int i = index_++;
195  for(unsigned int n = 0; n < count_; ++n) {
196  if(queues_[(i + n) % count_].TryPush(work)){
197  return res;
198  }
199  }
200  queues_[i % count_].Push(work);
201 
202  return res;
203  }
204 
205  unsigned int get_num_threads(){
206  return count_;
207  }

209 private:
210  void Run(unsigned int i) {
211  while (true) {
212  std::function<void()> f;
213  for (unsigned n = 0; n != count_; ++n) {
214  if (queues_[(i + n) % count_].TryPop(f)) {
215  break;
216  }
217  }
218  if (!f && !queues_[i].Pop(f)) {
219  break;
220  }
221  f();
222  }
223  }

228  std::vector<NotificationQueue> queues_;
229 };
